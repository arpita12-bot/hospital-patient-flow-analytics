USE hospital_db;
---- Patient Demographics By Gender
CREATE VIEW vw_patientflow AS 
SELECT 
    p.gender,
    COUNT(DISTINCT f.fact_id) AS total_patients
FROM dbo.fact_tbl_patient_flow f 
JOIN dbo.dim_tbl_patient p ON f.patient_sk = p.surrogate_key
GROUP BY p.gender;

SELECT * FROM vw_patientflow;

--- Total Patient count over time
CREATE VIEW vm_patient_volume_trend AS
SELECT
    CAST(f.admission_time AS DATE) AS admission_date,
    p.gender,
    COUNT(DISTINCT f.fact_id) AS patient_count
FROM dbo.fact_tbl_patient_flow f
JOIN dbo.dim_tbl_patient p 
    ON f.patient_sk = p.surrogate_key
GROUP BY 
    CAST(f.admission_time AS DATE),
    p.gender;

SELECT * FROM vm_patient_volume_trend;


---Total Patients over department
CREATE VIEW vm_department_inflow AS
SELECT 
    d.department,
    p.gender,
    COUNT(CASE WHEN f.is_currently_admitted = 1 THEN f.fact_id END) AS patient_count
FROM dbo.fact_tbl_patient_flow f
JOIN dbo.dim_tbl_patient p ON f.patient_sk = p.surrogate_key
JOIN dbo.dim_tbl_department d ON f.department_sk = d.surrogate_key
GROUP BY p.gender, d.department;

SELECT * FROM vm_department_inflow;

---- Average treatment duration

CREATE VIEW vm_avg_treatment_duration AS 
SELECT
    d.department,
    p.gender,
    AVG(f.length_of_stay_hours) AS avg_treatment_duration
FROM dbo.fact_tbl_patient_flow f 
JOIN dbo.dim_tbl_patient p ON f.patient_sk = p.surrogate_key
JOIN dbo.dim_tbl_department d ON f.department_sk = d.surrogate_key
GROUP BY d.department,p.gender;

SELECT * FROM vm_avg_treatment_duration;

--- Bed Occupancy by Gender (Beds occupied in total)
CREATE VIEW vm_bed_occupancy AS
SELECT  
    p.gender,
    COUNT(CASE WHEN f.is_currently_admitted = 1 THEN f.bed_id END) * 1.0 /COUNT(f.bed_id) * 100 AS bed_occupancy_percent
FROM dbo.fact_tbl_patient_flow f
JOIN dbo.dim_tbl_patient p ON f.patient_sk = p.surrogate_key
GROUP BY p.gender;

SELECT * FROM vm_bed_occupancy;


---- Total overstay patients count
CREATE VIEW vm_overstay_patients AS
SELECT
    d.department,
    p.gender,
    COUNT(f.fact_id) AS overstay_count
FROM dbo.fact_tbl_patient_flow f
JOIN dbo.dim_tbl_patient p ON f.patient_sk = p.surrogate_key
JOIN dbo.dim_tbl_department d ON f.department_sk = d.surrogate_key
WHERE f.length_of_stay_hours > 50
GROUP BY d.department, p.gender;

SELECT * FROM vm_overstay_patients;

--- Total bed turnover

CREATE VIEW vm_bed_turnover_rate as
SELECT
    p.gender,
    COUNT(DISTINCT f.fact_id) * 1.0 / COUNT(DISTINCT f.bed_id) AS bed_turnover_rate
FROM dbo.fact_tbl_patient_flow f
JOIN dbo.dim_tbl_patient p ON f.patient_sk = p.surrogate_key
GROUP BY p.gender;

SELECT * FROM vm_bed_turnover_rate;
